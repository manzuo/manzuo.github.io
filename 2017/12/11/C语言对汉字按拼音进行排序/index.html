struct student{
char name[10];
} stu[42];
int main(){
int n;
struct student s;
scanf(“%d”,&n);
for (int i=1;i<=n;i++){
scanf(“%s”,stu[i].name);
}
for (int i=1;i<=n-1;i++){
for(int j=1;j<=n-i;j++){
if(strcoll(stu[j].name,stu[j+1].name)>=0){
s=stu[j];
stu[j]=stu[j+1];
stu[j+1]=s;
}
}
}
printf(“\n\n”);
for(int i=1;i<=n;i++){
printf(“%s\n”,stu[i].name);
}
return 0;
}
